```typescript
import { useForm } from 'react-hook-form'
import { zodResolver } from '@hookform/resolvers/zod'
import { leadSchema, type LeadFormValues, SERVICE_TYPES } from '@/lib/validations/lead'
import { useState } from 'react'
```

**Props interface:**
```typescript
interface LeadFormProps {
  clientId: string      // hidden, passed from server component
  serviceType: string   // pre-filled from URL segment per D-19
}
```

**Component implementation:**
- Initialize `useForm<LeadFormValues>` with `zodResolver(leadSchema)` and defaultValues: `{ client_id: clientId, service_type: serviceType as LeadFormValues['service_type'], homeowner_name: '', phone: '', zip_code: '' }`
- Track submission state with `useState`: `isSubmitting` (boolean), `isSuccess` (boolean), `errorMessage` (string | null)
- On submit:
  1. Set `isSubmitting = true` (disables button per D-14 frontend idempotency)
  2. `fetch('/api/leads/submit', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) })`
  3. Parse response JSON
  4. If response.ok and body.success: set `isSuccess = true`
  5. If error: set `errorMessage = body.error || 'Something went wrong. Please try again.'`
  6. Finally: set `isSubmitting = false`

**Success state** (per D-15, LEAD-04): When `isSuccess` is true, render:
```
<div className="text-center py-8">
  <div className="text-green-600 text-4xl mb-4">✓</div>
  <h3 className="text-xl font-bold mb-2">Thank You!</h3>
  <p className="text-gray-600">We'll call you within 5 minutes.</p>
</div>
```

**Form fields** (per D-19 — exactly 4 visible fields):
1. **Name** — `<input type="text" placeholder="Your Name" {...register('homeowner_name')} />` with error display
2. **Phone** — `<input type="tel" placeholder="(555) 123-4567" {...register('phone')} />` with error display
3. **Service Type** — `<select {...register('service_type')}>` with `SERVICE_TYPES.map` as options. Pre-filled from URL but editable. Display labels: "AC Repair", "Furnace Repair", "HVAC Installation", "HVAC Maintenance"
4. **Zip Code** — `<input type="text" inputMode="numeric" maxLength={5} placeholder="Zip Code" {...register('zip_code')} />` with error display
5. **Hidden** — `<input type="hidden" {...register('client_id')} />`

**Submit button:**
```
<button type="submit" disabled={isSubmitting}
```
